/**
 * @file Product model
 * @description Defines the Mongoose schema for products in the database.
 */

const mongoose = require("mongoose");

/**
 * @typedef {Object} Product
 * @property {string} name - Name of the product (required)
 * @property {string} [description] - Optional description of the product
 * @property {number} price - Price of the product (required)
 * @property {string} [category] - Category the product belongs to
 * @property {number} [stock=0] - Number of items in stock
 * @property {string} [imageUrl] - URL of the product image
 * @property {boolean} [isDeleted=false] - Flag to indicate soft deletion (tombstone)
 * @property {Date} createdAt - Timestamp of creation (auto-generated)
 * @property {Date} updatedAt - Timestamp of last update (auto-generated)
 */

const productSchema = new mongoose.Schema({
  name: { type: String, required: true },
  description: String,
  price: { type: Number, required: true },
  category: String,
  stock: { type: Number, default: 0 },
  imageUrl: String,
  isDeleted: { type: Boolean, default: false }
}, { timestamps: true });

module.exports = mongoose.model("Product", productSchema);
